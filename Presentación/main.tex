\begin{frame}{Motivación y pregunta de investigación}
	
	\begin{block}{Pregunta inicial}
		¿Cómo emergen transiciones dinámicas de estabilidad en sistemas Lotka-Volterra generalizados de alta dimensión?
	\end{block}	
	En el modelo de Lotka-Volterra generalizado el Jacobiano en el equilibrio es en general
	$$\mathcal{J}_{ij}\propto x_i^*\alpha_{ij}$$
	por lo que la estabilidad depende explícitamente de las abundancias del equilibrio.
	
	\begin{alertblock}{Pregunta emergente}
		¿Cuál es la relación entre la \textbf{factibilidad}	del equilibrio y la \textbf{estabilidad dinámica} del sistema?
	\end{alertblock}
\end{frame}

\section{Modelo y planteamiento del problema}

\begin{frame}{Modelo}
	Se considera una comunidad ecológica de $N$ especies interactuando mediante el modelo de Lotka-Volterra generalizado
	$$r_ix_i\left (1-\frac{\sum_{j=1}^N\alpha_{ij}x_j}{K_i}\right ) = r_ix_i\left (1-\frac{\alpha_{ii}x_i}{K_i}\right )-r_ix_i\left (\frac{\sum_{j\neq i}\alpha_{ij}x_j}{K_i}\right )$$
	\begin{columns}
		\column{0.6\textwidth}
		\begin{itemize}
			\item $N$: número de especies.
			\item $\alpha_{ij}\in\Lambda$: interacciones entre especies.
			\item $r_i$: tasas intrínsecas de crecimiento.
			\item $K_i$: Capacidad de carga.
		\end{itemize}
		
		\column{0.4\textwidth}
		\begin{figure}
			\raggedright
			\includesvg[width=0.85\textwidth]{../Texto/Imagenes/RedIncidencias}
		\end{figure}
	\end{columns}
\end{frame}

\begin{frame}{Estructura de la red de interacciones}
	Sea un conjunto de $N$ nodos. Una red aleatoria se define por su matriz de adyacencia $\mathcal{E}\in\mathrm{M}_N(\{0,1\})$ tal que
	\newline
	\begin{columns}
		\column{0.49\textwidth}
		\textbf{Soporte simétrico}
		
		$$\mathcal{E}_{ij}=\begin{cases}
			X_{ij},\quad &i<j,\\
			X_{ji},\quad &i>j,\\
			0,\quad &i=j
		\end{cases},\quad X_{ij}\sim \text{Bernoulli}(p)$$
		
		
		\column{0.49\textwidth}
		\textbf{Soporte dirigido}
		
			$$
		\mathcal{E}_{ij}=\begin{cases}
			X_{ij},\quad &i\neq j,\\
			0,\quad &i=j
		\end{cases},\quad X_{ij}\sim\text{Bernoulli}(p)
		$$
		
	\end{columns}
	donde las variables $X_{ij}$ son independientes y cada enlace aparece con probabilidad $p$. Por tanto $\mathcal{E}$ puede ser simétrica como asimétrica.
\end{frame}

\begin{frame}{Red de interacciones ponderada}
	\begin{block}{Interacciones del modelo}
		$\alpha_{ij}:=$ representa el efecto de la especie $j$ sobre la especie $i$.
	\end{block}
	La red de interacciones se puede capturar en la \textbf{matriz de coeficientes de interacción} $\Lambda$ definida como
	$$\Lambda=(\mathcal{E}\odot\mathcal{M})+I$$
	\begin{itemize}
		\item $\mathcal{E}:$ red aleatoria con soporte dirigido o simétrico.
		\item $\mathcal{M}\in\mathrm{M}(\mathbb{R}):$ matriz con entradas i.i.d con media cero y varianza finita.
		\item $I:$ se introduce la matriz identidad como términos de auto-interacción que funcionan como mecanismo de regulación.
	\end{itemize}
\end{frame}

\begin{frame}{Tipo de interacciones: Clasifiación de May}
	\begin{columns}
		\column{0.6\textwidth}
		\begin{itemize}
			\item Cooperación $(++)$
			\item Competencia $(--)$
			\item Presa-depredador $(+-)$
			\item Comensalismo $(+0)$
			\item Amensalismo $(-0)$
		\end{itemize}
		\vspace{0.5cm}
		\begin{equation*}
			r_ix_i\left (1-\frac{\alpha_{ii}x_i}{K_i}\right )-r_ix_i\left (\frac{\sum_{j\neq i}\alpha_{ij}x_j}{K_i}\right )
		\end{equation*}

\begin{frame}{Equilibrios y factibilidad}
	Los estados estacionarios se obtienen imponiendo $\dot{x}_i=0$, lo que conduce a
	$$x_i^*=0,\quad\text{ó}\quad1+\frac{\sum_j\alpha_{ij}x_j^*}{K_i}=0$$
	cuando todo $x_i^*>0$ se obtiene,
	$$X^* = \Lambda^{-1}\mathrm{K},\qquad\mathrm{K}=(K_1,\dots,K_N)$$
	
	\vspace{0.3cm}
	\begin{block}{Factibilidad}
		Un equilibrio es físicamente relevante si $x_i^* \geq 0 \quad \forall i$
	\end{block}
	\vspace{0.3cm}
	\begin{columns}
		\column{0.5\textwidth}
		\textbf{Coexistencia}
		
		\[
		x_i^* > 0 \quad \forall i
		\]
		
		Todas las especies sobreviven.
		
		\column{0.5\textwidth}
		\textbf{Equilibrio parcial}
		
		Algunas especies se extinguen:
		\[
		x_i^* = 0
		\]
		
		La dinámica efectiva ocurre en un subespacio.
	\end{columns}
\end{frame}

\begin{frame}{Estructura del Jacobiano}
	En el equilibrio $\X$, la matriz Jacobiana toma la forma
	$$\mathcal{J}_{ij}=-\frac{r_ix_i^*}{K_i}\, \alpha_{ij}$$
	\vspace{0.3cm}
	o equivalentemente
	$$\mathcal{J}=-D(\X)\, \Lambda$$
	donde $D(\X)=\operatorname{diag}\left (\frac{r_ix_i^*}{K_i}\right )$
	\vspace{0.4cm}
	\begin{block}{Observación clave}
		La estabilidad depende explícitamente de las abundancias del equilibrio.
	\end{block}
\end{frame}

\begin{frame}{Factibilidad e invasibilidad}
	\begin{columns}
		\column{0.5\textwidth}
		\textbf{Coexistencia}
		
		$$x_i^*>0,\quad\forall i$$
		
		Todas las especies contribuyen al Jacobiano.
		
		\column{0.5\textwidth}
		\textbf{Extinciones}
		
		$$x_i^*=0$$
		
		Se anulan filas del Jacobiano.
	\end{columns}
	\vspace{0.4cm}
	\begin{alertblock}{Invasibilidad}
		Para especies extintas, la tasa 
		$$r_i\left (1-\frac{\sum_j\alpha_{ij}x_j^*}{K_i}\right )$$
		determina si puede invadir el equilibrio, introduciendo un mecanismo de inestabilidad.
	\end{alertblock}
\end{frame}

\section{Resultados}

\begin{frame}{Criterio clásico de estabilidad}
	\begin{columns}
		\column{0.3\textwidth}
		\textbf{Criterio de estabilidad:} $$\sigma\sqrt{NC}<d$$
		
		\vspace{0.3cm}
		Parámetros utilizados: 
		\begin{itemize}
			\item $N=100$
			\item $\sigma=0.4$
			\item $p=\frac{1}{\sigma^2 N}-0.03$
		\end{itemize}
		
		\column{0.7\textwidth}
		
		\begin{figure}
			\centering
			\includegraphics[scale=0.13]{../Texto/Imagenes/LeyCircularMay2}
		\end{figure}
		
	\end{columns}
	\vspace{0.3cm}
	La estabilidad se determina por propiedades estadísticas de $A$. No incorpora la dependencia del sistema en $\X$.
\end{frame}

\begin{frame}{Criterio suficiente de estabilidad en el modelo LV}
	Los valores propios de $\mathcal{J}$ están contenidos en discos
	$$\lambda(\mathcal{J})\subset\bigcup_i D(\mathcal{J}_{ii},R_i),\quad R_i=\sum_{j\neq i}|\mathcal{J}_{ij}|$$
	\vspace{0.3cm}
	\begin{block}{Condición suficiente}
		$$\mathcal{J}_{ii}+\sum_{j\neq i}|\mathcal{J}_{ij}|<0\qquad\forall i$$
	\end{block}	
	\vspace{0.3cm}
	\centering
	La estabilidad depende del balance ente términos diagonales e interacciones.

\end{frame}

\begin{frame}{Criterio suficiente de estabilidad en el modelo LV}
	En el modelo,
	$$\mathcal{J}=-D(\X)\Lambda$$
	
	\vspace{0.3cm}
	
	Bajo supuestos estadísticos 
	\[
	\mathbb{E}\left[\sum_{j\neq i} |J_{ij}|\right]
	\sim (N-1)\,p\,\sigma \sqrt{\frac{2}{\pi}}
	\]
	\vspace{0.4cm}
	
	\begin{alertblock}{Interpretación}
		La estabilidad está determinada por la competencia entre
		\[
		|J_{ii}| \quad \text{y} \quad (N-1)\,p\,\sigma \sqrt{\frac{2}{\pi}}
		\]
	\end{alertblock}
	
	
\end{frame}

\begin{frame}{Factibilidad del equilibrio}
	\begin{columns}
		\column{0.6\textwidth}
		Las abundancias de un sistema ecológico deben satisfacer 
		$$\X\geq 0 \quad \forall i$$
		\begin{block}{Problema}
			Dado el sistema 
			$$\Lambda\X=\mathrm{K}$$
			¿para qué condiciones sobre los parámetros es factible?
		\end{block}
		
		\vspace{0.3cm}
		
		Esta condición define una región geométrica en $\mathbb{R}^n$ que puede caracterizarse como un \textbf{cono convexo}.
		
		\column{0.4\textwidth}
		\includegraphics[scale=0.08]{../../Tesis/Texto/Imagenes/Convex_cone_illust.svg.png}
		\small¡Factibilidad no implica estabilidad!
	\end{columns}
	 
\end{frame}

\begin{frame}{Factibilidad del equilibrio}
	\begin{columns}
		\column{0.6\textwidth}
			\begin{block}{Cono convexo}
			Combinaciones positivas de vectores columna $\ell_i\in\Lambda$:
			$$\operatorname{C}(\Lambda):=\left \{\sum_{i=1}^N  x_i\ell_i : x_i>0\right \}$$
		\end{block}
		\vspace{0.3cm}
		Si $\mathrm{K}\in\operatorname{C}(\Lambda)$ entonces $X^*>0$. Para ello se estudiaron dos esquemas:
		\vspace{0.15cm}
		\begin{itemize}
			\item Diagonal dominante
			\item Relación señal-ruido
		\end{itemize}
		
		\column{0.4\textwidth}
		\includegraphics[scale=0.08]{../../Tesis/Texto/Imagenes/Convex_cone_illust.svg.png}
	\end{columns}

\end{frame}

\begin{frame}{Factibilidad del equilibrio}
	\begin{columns}
		\column{0.48\textwidth}
		\textbf{Diagonal dominante}
		
		\vspace{0.2cm}
		
		Se impone
		
		$$\sum_{j\neq i}|\alpha_{ij}|<|\alpha_{ii}|$$
		
		\vspace{0.2cm}
		
		lo que conduce a la condición suficiente
		
		$$(N-1)p\sigma\sqrt{\frac{2}{\pi}}<1$$
		
		\vspace{0.2cm}
		
		\textit{Implica estabilidad y factibilidad, pero es altamente restrictiva.}
		
		\column{0.52\textwidth}
		\textbf{Relación señal--ruido (SNR)}
		
		\vspace{0.2cm}
		
		Se busca relajar la dominancia diagonal y caracterizar el equilibrio en términos estadísticos.
		
		\vspace{0.2cm}
		
		
		$$\kappa=\frac{\mu}{\sigma_{\text{eff}}}$$
		
		\vspace{0.2cm}
		
		\begin{itemize}
			\item $\mu$: término determinista (autointeracción)
			\item $\sigma_{\text{eff}}$: fluctuaciones colectivas
		\end{itemize}
		
		\vspace{0.2cm}
		
		\textit{Controla la transición entre regímenes factibles e inviables.}
	\end{columns}
\end{frame}

\begin{frame}{Parámetro de factibilidad}
	
	La factibilidad depende del balance entre:
	
	$$\text{interacciones medias} \quad \text{vs} \quad \text{fluctuaciones}$$
	
	\vspace{0.3cm}
	
	Bajo supuestos estadísticos,
	
	$$Y \sim \mathcal{N}(1, (N-1)p\sigma^2)$$
		
	\vspace{0.3cm}
	
	\begin{block}{Parámetro efectivo}
		$$\kappa = \frac{1}{\sigma\sqrt{Np}}$$
		considerando que $N-1\approx N $
	\end{block}
	
\end{frame}

\begin{frame}{Probabilidad de factibilidad}
	
	La condición de factibilidad requiere
	
	\[
	Y > 0
	\]
	
	\vspace{0.3cm}
	
	Por lo tanto,
	
	\[
	\text{Pr}(Y > 0) = \Phi(\kappa)
	\]
	
	\vspace{0.3cm}
	
	\begin{alertblock}{Interpretación}
		La factibilidad depende únicamente del parámetro $\kappa$.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Umbral de transición}
	Cuando mayor es el valor de $\kappa$, los términos de auto-interacción son más dominantes.
	
	\vspace{0.3cm}
	
	\begin{block}{Pregunta}
		¿Qué umbral de $\kappa$ es necesario para distinguir la factibilidad del sistema?
	\end{block}
	\vspace{0.3cm}
	
	Cuando $p=\sigma=1.0$, entonces $\kappa=0.1005$ y $\Phi(\kappa)\approx 0.54$
	\vspace{0.3cm}
	\begin{exampleblock}{$\kappa$ crítica}
		Cuando $\kappa\approx0.9$, $\Phi(\kappa)=0.8159$ y además $\frac{1}{\kappa}\approx 1.1111\dots$
	\end{exampleblock}
	
\end{frame}
